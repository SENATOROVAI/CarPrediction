from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity
import glob

# Модель, которая работает с изображениями
model = SentenceTransformer('clip-ViT-B-32')

def get_image_embeddings(image_paths):
    """Получение векторных представлений изображений"""
    embeddings = model.encode([image_path for image_path in image_paths])
    return embeddings

def find_similar_images_clip(query_image_path, image_paths, top_k=5):
    """Поиск похожих изображений с помощью CLIP"""
    all_paths = [query_image_path] + image_paths
    embeddings = get_image_embeddings(all_paths)
    
    query_embedding = embeddings[0]
    other_embeddings = embeddings[1:]
    
    similarities = cosine_similarity([query_embedding], other_embeddings)[0]
    
    results = list(zip(image_paths, similarities))
    results.sort(key=lambda x: x[1], reverse=True)
    
    return results[:top_k]

import os

from PIL import Image
from collections import defaultdict
import cv2

def find_duplicate_cars(image_folder):
    """Найти дубликаты фотографий машин"""
    # Словарь для хранения хэшей и путей к файлам
    hash_dict = defaultdict(list)
    
    # Поддерживаемые форматы
    extensions = {'.jpg', '.jpeg', '.png', '.bmp', '.webp'}
    
    print("Анализируем изображения...")
    
    for filename in os.listdir(image_folder):
        if any(filename.lower().endswith(ext) for ext in extensions):
            filepath = os.path.join(image_folder, filename)
            
            try:
                # Загружаем и вычисляем хэш
                with Image.open(filepath) as img:
                    # Используем pHash (перцептивный хэш)
                    img_hash = imagehash.phash(img)
                    
                # Добавляем в словарь
                hash_dict[str(img_hash)].append(filename)
                
            except Exception as e:
                print(f"Ошибка обработки {filename}: {e}")
    
    # Находим дубликаты
    duplicates = {hash_val: files for hash_val, files in hash_dict.items() 
                 if len(files) > 1}
    
    return duplicates

# Использование

import os
import imagehash
from PIL import Image
import cv2
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity
from collections import defaultdict

class AdvancedCarMatcher:
    def init(self):
        self.methods = ['phash', 'color_histogram']
    
    def compute_phash(self, image_path):
        """Вычисление перцептивного хэша"""
        with Image.open(image_path) as img:
            return imagehash.phash(img, hash_size=16)
    
    def compute_color_histogram(self, image_path, bins=64):
        """Вычисление цветовой гистограммы"""
        image = cv2.imread(image_path)
        if image is None:
            return None
            
        # Конвертируем в HSV
        hsv = cv2.cvtColor(image, cv2.COLOR_BGR2HSV)
        
        # Вычисляем гистограмму для канала Hue
        hist = cv2.calcHist([hsv], [0], None, [bins], [0, 180])
        cv2.normalize(hist, hist)
        
        return hist.flatten()
    
    def find_duplicate_cars(self, image_folder):
        """Поиск дубликатов машин несколькими методами"""
        files = [f for f in os.listdir(image_folder) 
                if f.lower().endswith(('.jpg', '.jpeg', '.png'))]
        
        results = defaultdict(list)
        
        print(f"Обрабатываем {len(files)} изображений...")
        
        # Собираем все хэши и гистограммы
        hashes = []
        histograms = []
        valid_files = []
        
        for filename in files:
            filepath = os.path.join(image_folder, filename)
            
            try:
                phash = self.compute_phash(filepath)
                hist = self.compute_color_histogram(filepath)
                
                if hist is not None:
                    hashes.append(phash)
                    histograms.append(hist)
                    valid_files.append(filename)
                    
            except Exception as e:
                print(f"Ошибка обработки {filename}: {e}")
        
        # Поиск дубликатов по pHash
        print("Поиск дубликатов по перцептивному хэшу...")
        for i, hash1 in enumerate(hashes):
            group = [valid_files[i]]
            
            for j, hash2 in enumerate(hashes[i+1:], i+1):
                if hash1 - hash2 <= 8:  # Пороговое значение
                    group.append(valid_files[j])
            
            if len(group) > 1:
                results['phash'].append(group)
        
        # Поиск по цветовым гистограммам
        print("Поиск по цветовым гистограммам...")
        for i, hist1 in enumerate(histograms):
            group = [valid_files[i]]
            
            for j, hist2 in enumerate(histograms[i+1:], i+1):
                # Косинусное сходство гистограмм
                similarity = cosine_similarity([hist1], [hist2])[0][0]
                if similarity > 0.85:  # Порог схожести
                    group.append(valid_files[j])
            
            if len(group) > 1:
                results['color'].append(group)
        
        return results

# Использование

for method, groups in results.items():
    print(f"\nРезультаты метода '{method}':")
    for i, group in enumerate(groups, 1):
        print(f"  Группа {i}: {group}")
